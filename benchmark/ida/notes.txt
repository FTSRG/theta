library(ggplot2)
library(dplyr)
library(iplots)
library(party)
d <- read.csv("data/log_20161206_154049.csv", sep=",")
# check column type
str(d)
# check summary values
summary(d)
# iplots interactive
ibar(d$Model)
ibar(d$Safe)
ibox(d$Time..ms.)
# filtering, grouping
d %>% filter(Model == "hw/shortp0.aag") %>% group_by(Domain) %>% summarize(median(Time, na.rm=TRUE))
# plotting
qplot(data=d, Safe, CEX.length, geom="boxplot")
qplot(data=d, Iterations, ARG.size, color=Domain) + facet_wrap(~Model)
# decision tree
d.no.to <- d %>% filter(! is.na(Time))
tree <- ctree(  Time ~ Model + Domain + Refinement + Search, data = d.no.to)
plot(tree)
